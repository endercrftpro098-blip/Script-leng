# Script-leng
proyecto de bash para ejecutar scripts de multiples lenguajes en linux
#Guia de descarga
git clone https://github.com/endercrftpro098-blip/Script-leng.git
cd ~/Scripts-leng
chmod +x install.sh
./install.sh
#Uso
Al tener todo el script se va a /usr/local/bin
luego para probarlo debes poner "proyectos" en la consola
y te muestra los lenguajes disponibles
